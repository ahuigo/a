        "fmt"
        "os"
    )

    func main() {
        pwd, err := os.Getwd()
        if err != nil {
            fmt.Println(err)
            os.Exit(1)
        }
        fmt.Println(pwd)
    }

### sync

    f.Sync()

    // `bufio` provides buffered writers in addition
    // to the buffered readers we saw earlier.

    // Use `Flush` to ensure all buffered operations
